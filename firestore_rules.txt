
rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // Helper function to check for admin role safely.
    function isAdmin() {
      return request.auth != null && 
             exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // Helper to check if user is authenticated and matches the userId
    function isOwner(userId) {
      return request.auth != null && request.auth.uid == userId;
    }

    // Site Settings: Publicly readable for everyone. Writable only by admins.
    match /site_settings/{document} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // Subscription Plans: Publicly readable. Writable only by admins.
    match /subscription_plans/{planId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // Services: Readable by any authenticated user, writable only by admins.
    // Allow updating usageCount by authenticated users.
    match /services/{serviceId} {
      allow read: if request.auth != null;
      allow create, delete: if isAdmin();
      allow update: if isAdmin() || (request.auth != null && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['usageCount']));
    }

    // Users collection
    match /users/{userId} {
      // Admins can read all users. Owners can read their own.
      allow read: if isOwner(userId) || isAdmin();
      
      // Owners can create their own doc (signup). Admins can create any.
      allow create: if isOwner(userId) || isAdmin();
      
      // Owners can update their own doc. Admins can update any.
      allow update: if isOwner(userId) || isAdmin();
      
      // Only admins can delete users.
      allow delete: if isAdmin();

      match /subscriptions/{subscriptionId} {
        allow read: if isOwner(userId) || isAdmin();
        allow write: if isAdmin();
      }

      match /checkout_sessions/{sessionId} {
        allow read, write, create: if isOwner(userId);
      }
    }

    // Customers collection (used by the Stripe extension)
    match /customers/{userId} {
      // Allow Admin to read/write customer docs to support subscription management
      allow read, write: if isOwner(userId) || isAdmin();

      // Explicit rule for subscriptions created by Stripe extension
      match /subscriptions/{subscriptionId} {
         allow read: if isOwner(userId) || isAdmin();
         allow write: if isAdmin(); // Only admins (or the extension via Admin SDK) should write manually
      }
      
      // Rules for creating checkout sessions
      match /checkout_sessions/{sessionId} {
         allow read, write, create: if isOwner(userId);
      }
      
      // Rules for payments
      match /payments/{paymentId} {
         allow read: if isOwner(userId) || isAdmin();
      }
    }

    // Support Tickets System
    match /support_tickets/{ticketId} {
      // Users can create tickets (assigning themselves as userId)
      allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
      
      // Users can read/update their own tickets. Admins can read/update all.
      allow read, update: if request.auth != null && (resource.data.userId == request.auth.uid || isAdmin());
      
      // Only admins can delete tickets
      allow delete: if isAdmin();

      // Messages subcollection
      match /messages/{messageId} {
         // Allow read/create if user owns the parent ticket or is admin
         allow read, create: if request.auth != null;
      }
    }

    // Stripe extension's collections for products and prices
    match /products/{productId} {
        allow read: if true;
        allow write: if isAdmin();

        match /prices/{priceId} {
            allow read: if true;
            allow write: if isAdmin();
        }
        match /roles/{roleId} {
            allow read: if true;
            allow write: if isAdmin();
        }
    }
  }
}
